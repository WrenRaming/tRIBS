52/*
53 A simple compliant implementation of auto_ptr.
54 This is from Greg Colvin's implementation posted to comp.std.c++.
55
56 Instead of using mutable this casts away const in release.
57
58 We have to do this because we can't build containers of these
59 things otherwise.
60 */
62
63template<class X>
64class my_auto_ptr
65{
66 X* px;
67public:
68 my_auto_ptr() : px(0) {}
69 my_auto_ptr(X* p) : px(p) {}
70 my_auto_ptr(const my_auto_ptr<X>& r) : px(r.release()) {}
71 my_auto_ptr& operator=(const my_auto_ptr<X>& r)
72 {
73 if (&r != this)
74 {
75 delete px;
76 px = r.release();
77 }
78 return *this;
79 }
80 ~my_auto_ptr() { delete px; }
81 X& operator*() const { return *px; }
82 X* operator->() const { return px; }
83 X* get() const { return px; }
84 X* release() const { X *p=px; ((my_auto_ptr<X>*)(this))->px=0; return p; }
85};
