156int tMeshList< NodeType >::
157removeNext( NodeType &value, tListNode< NodeType > * ptr )
158{
159 if( ptr->next == 0 ) return 0;
160 if( ptr == 0 ) return 0;
161 if( ptr->next == lastactive ) return removeFromActiveBack( value );
162 if( ptr == lastactive ) return removeFromBoundFront( value );
163 if( tList< NodeType >::removeNext( value, ptr ) ){
164 int act = value.getBoundaryFlag();
165 if ((act == kNonBoundary) || (act == kStream )) nActiveNodes--;
166 return 1;

172template< class NodeType >
173int tMeshList< NodeType >::
174nextToBack( tListNode< NodeType > * prev )
175{
176 // Check if there was a previous node given and
177 // that there is a next node
178 if( prev == 0 ) return 0;
179 if( prev->next == 0 ) return 0;
180 if (prev == this->lastactive) return 0;
181 // This is the node that will be moved
182 tListNode< NodeType > *nnode = prev->next;
183 nActiveNodes--;
184 // Hook up previous to the next node
185 prev->next = nnode->next;
186 // Check if last active and make previous so
187 if (this->lastactive == nnode) this->lastactive = prev;
188 // Hook up moved node to end
189 this->last->next = nnode;
190 nnode->next = 0;
191 this->last = nnode;
192 return 1;
193}

196template< class NodeType >
197int tMeshList< NodeType >::
198frontToBack()
199{
200 // Check if list is empty
201 if ( this->isEmpty() ) return 0;
202 // This is the node to be moved
203 tListNode< NodeType > *nnode = this->first;
204 nActiveNodes--;
205 // If lastactive, set to 0
206 if (nnode == this->lastactive) this->lastactive = 0;
207 // If no other node, done
208 if (nnode->next == 0) return 1;
209 // Make the next node first
210 this->first = nnode->next;
211 // Move node to end
212 this->last->next = nnode;
213 nnode->next = 0;
214 this->last = nnode;
215 return 1;
216}

218template< class NodeType >
219int tMeshList< NodeType >::
220removeFromBoundFront( NodeType &value )
221{
222 assert( &value != 0 );
223 if( this->isEmpty() ) return 0;
224 else if( this->last == lastactive ) return 0;
225 else{
226 tListNode< NodeType > * temp = lastactive->next;
227 if( this->first == this->last ) this->first = this->last = 0;
228 else lastactive->next = lastactive->next->next;
229 value = temp->data;
230 delete temp;
231 this->nNodes--;
232 return 1;
233 }

236template< class NodeType >
237int tMeshList< NodeType >::
238removeFromActiveBack( NodeType &value )
239{
240 if( this->isEmpty() ) return 0;
241 else{
242 tListNode< NodeType > * temp = lastactive;
243 if( this->first == lastactive ) lastactive = 0;
244 if( this->first == this->last ) this->first = this->last = 0;
245 else{
246 tListNode< NodeType > * current = this->first;
247 while( current->next != lastactive ) current = current->next;
248 current->next = lastactive->next;
249 lastactive->next = 0;
250 lastactive = current;
251 }
252 value = temp->data;
253 delete temp;
254 this->nNodes--;
255 nActiveNodes--;
256 return 1;
257 }
