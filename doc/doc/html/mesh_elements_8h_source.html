834inline double tEdge::CalcSlope(){
835 assert( org!=0 ); // Failure = edge has no origin and/or destination node
836 assert( dest!=0 );
837 assert( len>0.0 );
838 slope = ( org->getZ() - dest->getZ() ) / len;

857inline double tEdge::CalcVEdgLen(){
858 assert( ccwedg!=0 );
859 double dx, dy;
860
861 dx = rvtx[0] - ccwedg->rvtx[0];
862 dy = rvtx[1] - ccwedg->rvtx[1];
863 vedglen = sqrt( dx*dx + dy*dy );
864 return( vedglen );
865}
